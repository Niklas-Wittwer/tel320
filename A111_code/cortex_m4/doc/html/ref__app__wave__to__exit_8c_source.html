82 int acc_ref_app_wave_to_exit(int argc, char *argv[])
83 {
84  (void)argc;
85  (void)argv;
86 
87  printf("Acconeer software version %s\n", acc_version_get());
88 
89  const acc_hal_t *hal = acc_hal_integration_get_implementation();
90 
91  // Timing
92  uint32_t period_length_ms = 1000U / UPDATE_RATE_HZ;
93 
94  if (!acc_rss_activate(hal))
95  {
96  printf("Failed to activate RSS\n");
97  return EXIT_FAILURE;
98  }
99 
100  acc_detector_presence_configuration_t configuration = acc_detector_presence_configuration_create();
101 
102  if (configuration == NULL)
103  {
104  printf("Failed to create detector configuration\n");
105  acc_rss_deactivate();
106  return EXIT_FAILURE;
107  }
108 
109  configure_detector(configuration);
110 
111  acc_detector_presence_handle_t handle = acc_detector_presence_create(configuration);
112 
113  if (handle == NULL)
114  {
115  printf("Failed to create detector\n");
116  acc_detector_presence_configuration_destroy(&configuration);
117  acc_detector_presence_destroy(&handle);
118  acc_rss_deactivate();
119  return EXIT_FAILURE;
120  }
121 
122  acc_detector_presence_result_t result;
123 
124  bool status = true;
125  bool cool_trig = true;
126  bool cool_time = true;
127  uint32_t cool_counter = 0;
128 
129  uint32_t last_update_ms = 0;
130 
131  status = acc_detector_presence_activate(handle);
132 
133  while (status)
134  {
135  while (last_update_ms != 0 && hal->os.gettime() - last_update_ms < period_length_ms)
136  {
137  acc_integration_sleep_ms(period_length_ms - (hal->os.gettime() - last_update_ms));
138  }
139 
140  status = acc_detector_presence_get_next(handle, &result);
141  last_update_ms = hal->os.gettime();
142 
143  if (status)
144  {
145  bool wave_to_exit = false;
146 
147  // Trigger "Wave to exit" if presence is detected and cool-down citerias has been met.
148  if (result.presence_detected && cool_trig && cool_time)
149  {
150  wave_to_exit = true;
151  cool_trig = false;
152  cool_time = false;
153  }
154 
155  // The first cool-down criteria is that the presence score must fall below a threshold
156  // before a new "Wave to exit" is triggered
157  if (!cool_trig && (result.presence_score < COOL_DOWN_THRESHOLD))
158  {
159  cool_trig = true;
160  cool_counter = COOL_DOWN_TIME_TICKS;
161  }
162 
163  // The second cool-down criteria is that a certain time, if set, must elapse before
164  // a new "Wave to exit" is triggered
165  if (!cool_time && cool_trig)
166  {
167  if (cool_counter > 0)
168  {
169  cool_counter -= 1;
170  }
171 
172  if (cool_counter == 0)
173  {
174  cool_time = true;
175  }
176  }
177 
178  if (wave_to_exit)
179  {
180  printf("Wave detected\n");
181  }
182  else
183  {
184  printf("No wave detected\n");
185  }
186  }
187  }
188 
189  // We will never exit so no need to destroy the configuration or detector
190 
191  return status ? EXIT_SUCCESS : EXIT_FAILURE;
192 }
